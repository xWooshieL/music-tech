\begin{frame}{Baseline 1: OLTW \small(Dixon 2005)}
  \small
  \textbf{On-Line Time Warping} --- инкрементальный вариант DTW.

  \begin{columns}[T,onlytextwidth]
    \column{0.42\textwidth}
    \vspace{0.4em}
    \begin{itemize}\setlength{\itemsep}{0.45em}
      \item Локальная стоимость:\\
            $\displaystyle c(i,t) = |\,p_i - o_t|$.
      \item Рекурсия:\\
            $\displaystyle D[i,t] = c(i,t) + \min\bigl(D[i{-}1,t{-}1],$\\
            \mbox{}\hfill$D[i{-}1,t],\, D[i,t{-}1]\bigr)$.
      \item $\hat\imath_t = \argmin_i D[i,t]$.
      \item Память $O(N)$ --- хранятся 2 колонки.
      \item \emph{RunCount} fail-safe: $k$ событий без сдвига
            $\Rightarrow$ форсируем шаг.
    \end{itemize}

    \column{0.58\textwidth}
    \centering
    \includegraphics[width=\linewidth]{figures/png/oltw_dp.png}
  \end{columns}

  \vspace{0.4em}
  {\footnotesize \textbf{Плюсы:} простота, устойчивость к rubato,
   всегда выдаёт ответ.\quad
   \textbf{Минусы:} нет confidence, плохо ловит структурные ошибки.}
\end{frame}
